# Question 1: Historgram equivilization using numpy for a simple image

img_gray = np.mean(img, axis=2)

hist, bins = np.histogram(img_gray.flatten(), 256, [0, 256])

cdf = hist.cumsum()

cdf_normalized = (cdf - cdf.min()) * 255 / (cdf.max() - cdf.min())

img_equalized = np.interp(img_gray, bins[:-1], cdf_normalized)

plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.imshow(img_gray, cmap='gray')
plt.subplot(1, 2, 2)
plt.imshow(img_equalized, cmap='gray')
plt.show()

# Question 2: Rotating an Image

img_grayscale = np.mean(img, axis=2)

rotation_angle = 45

rotation_angle_radians = rotation_angle * np.pi / 180

sin_rotation_angle = np.sin(rotation_angle_radians)
cos_rotation_angle = np.cos(rotation_angle_radians)

img_rotated = np.zeros_like(img)

for y in range(img.shape[0]):
    for x in range(img.shape[1]):

        new_x = int(x * cos_rotation_angle - y * sin_rotation_angle)
        new_y = int(x * sin_rotation_angle + y * cos_rotation_angle)


        if 0 <= new_x < img.shape[1] and 0 <= new_y < img.shape[0]:

            img_rotated[new_y, new_x] = img[y, x]

plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.imshow(img, cmap='gray')
plt.subplot(1, 2, 2)
plt.imshow(img_rotated, cmap='gray')
plt.show()
